import itchat
import threading
import re
from queue import Queue

# 创建一个全局的消息队列，用来暂存微信的数据
g_q = Queue()


def save_2_file(msg):
    """
    将接收到的数据写入到文件中
    """
    with open("wechat_撤销消息记录.txt", "a") as f:
        f.write("\r\n")  # 先与之前的内容用一个空行进行分割
        # 昵称
        NickName = msg['User']['NickName']
        # 备注名称
        RemarkName = msg['User']['RemarkName']
        # 获取撤销的内容
        Content = msg['Content']
        # 写入到文件
        f.write("%s(%s):%s" % (NickName, RemarkName, Content))


def handle_revoke_msg():
    """
    对接收的微信的数据进行处理，然后根据自己设定的逻辑进行数据处理
    """
    # 定义字典，用来存储运行后 接收到了所有的信息消息
    msg_dict = dict()

    while True:
        # 接收消息
        msg = g_q.get(True)
        print("接收到msg", type(msg), msg)

        if msg['Type'] == "Text":
            # 文本消息
            # 提取msg_id
            msg_id = msg['MsgId']
            # 将接收的消息添加到字典中
            msg_dict[msg_id] = msg  # 让整个msg字典，当做value，让msg的id当做key，这样更方便处理
        elif msg['Type'] == "Note":
            # 通知消息，例如 撤销
            # 提取msg_id
            # ----------- 正则表达式    开始 （我们还没有学习，到底2学期再学）----------
            ret = re.search(r"<msgid>([^<]+)", msg['Content'])
            if ret:
                revoke_msg_id = ret.group(1)
            # ----------- 正则表达式    结束     ----------
                print("刚刚撤销的消息id为：%s" % revoke_msg_id)
                del_msg = msg_dict.get(revoke_msg_id)
                if del_msg:
                    save_2_file(del_msg)

        print(">" * 30)
        print("字典现有数据为:\n")
        print(msg_dict)
        print("\n" * 3)

def main():
    """
    主控制程序
    """

    # 创建一个新的线程，用来接收从微信中收到消息后进行的处理
    t2 = threading.Thread(target=handle_revoke_msg)
    t2.setDaemon(True)  # 设置为守护线程，即当主线程结束后 也让这个子线程结束
    t2.start()


    # 主线程接收微信的文本数据
    itchat.auto_login(hotReload=True)
    itchat.run()


if __name__ == '__main__':
