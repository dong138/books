OpenCV基础

服务器开发-单任务版

1. 说明

  1. 单任务版，功能不复杂
  2. 主要任务：
    1. 接收客户端的链接，然后接收其发送过来的数据
    2. 首先要知道客户端发送过来的一张图片数据有多少，即先接收8个字节，提取出后续这个图片的数据大小
    3. 根据上一步提取出来的大小，接收这一张图片的数据
    4. 解压缩、将BGR颜色转换为RGB
    5. 使用pygame显示

2. 参考代码

import socket
import cv2
import struct
import numpy
import time
import pygame
from pygame.locals import *


class RemoteMonitorServer(object):
    def __init__(self, local_port):
        # 创建tcp服务器套接字
        self.server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.server_socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        self.server_socket.bind(("", local_port))
        self.server_socket.listen(128)

        # pygame的初始化代码
        pygame.init()
        # 创建一个窗口
        self.screen = pygame.display.set_mode((640, 480), 0, 32)


    def run(self):
        while True:
            client_socket, client_info = self.server_socket.accept()

            while True:
                # 接收8个字节，来得到接下来接收的这张图片的数据量大小
                recv_content = client_socket.recv(8)
                if not recv_content:
                    # 如果对方调用了close，那么此时这里的recv_content变量的数据长度为0
                    client_socket.close()
                    break

                info = struct.unpack("ihh", recv_content)
                # 提取大小
                buf_size = info[0]  # 获取读的图片总长度
                try:
                    buf = b""  # 代表bytes类型
                    # 接收完这张图片的数据(3)
                    while buf_size > 0:
                        temp_buf = client_socket.recv(buf_size)
                        buf_size -= len(temp_buf)
                        buf += temp_buf  # 将刚刚接收到的数据 加在一起

                    # 因为从网络中接收到的数据是将图片进行压缩之后的数据，所以接下来需要先解压，然后才能进一步处理
                    # (1) 将bytes类型数据转换成 numpy数组类型
                    data = numpy.fromstring(buf, dtype='uint8')

                    # (2) 进行解码
                    img = cv2.imdecode(data, cv2.IMREAD_COLOR)

                    # 如下代码非常重要，因为opencv采用的是bgr顺序存储一个像素
                    # 而pygame中是采用的rgb，所以需要进行转换
                    # (3) 将（OpenCV中）BGR-->(pygame中)RGB
                    img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)

                    # 因为此时图片不是本地硬盘上的图片，而是网络接收到的数据
                    # 所以需要使用pygame中frombuffer来加载这个数据，认为是一个图片
                    img = pygame.image.frombuffer(img, (640, 480), "RGB")
                    # 将这个图片粘贴到窗口中进行显示
                    self.screen.blit(img, (0, 0))
                    pygame.display.update()

                    # 检测是否点击关闭
                    for event in pygame.event.get():
                        # 判断是否是点击了退出按钮
                        if event.type == QUIT:
                            print("exit")
                            exit()  # 让整个程序结束，return让函数结束，break让循环结束

                except Exception as ret:
                    print("产生异常")
                    print(ret)

    def __del__(self):
        self.server_socket.close()


if __name__ == '__main__':

    # 1. 询问本地port
    local_port = input("请输入本地使用的port:")
    if not local_port:
        local_port = 8800
    else:
        local_port = int(local_port)

    # 2. 创建"远程监控"服务对象
    server = RemoteMonitorServer(local_port)

    # 3. 调用run方法
    server.run()

3. 总结

  1. cv2解压缩的过程，工作中基本都是查询资料得知，并不需要背，只要知道怎样查询即可
  2. 为了描述这个图片的大小，客户端发送过来的数据中有2部分组成
    1. 信息头：一张图片的大小，以及宽、高（虽然我们没有使用，但是如果想扩展业务这个功能可能需要）
    2. 图片数据：这一张图片真正的数据
