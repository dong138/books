Go基础

结构体

1. 引入

现在有一个需求，要求存储学生的详细信息，例如，学生的学号，学生的姓名，年龄，家庭住址等。按照以前学习的存储方式，

可以以如下的方式进行存储：

package main

import "fmt"

func main()  {
    var id int
    var name string
    var sex bool
    var age int
    var address string
    var weight int
}

以上方式，比较麻烦，并且不利于数据的管理。

为了便于对某个需要多种类型数据存储的情况，我们可以考虑结构体

2. 结构体

结构体：可以完成将多个数据类型的变量组合成一个整体来表示数据

3. 定义结构体类型

基本格式

type 结构体类型名 struct{
    成员变量名1 类型1
    成员变量名2 类型2
    成员变量名3 类型3
    ...
}

注意：

  • 大括号中是结构体的成员，注意在定义结构体成员时，不要加var

4. 定义结构体变量

方式1

// 定义结构体变量时，一定要进行初始化
var xiaomign Student = Student{101, "dongge", true, 18, "shandong", 100}

方式2

// 定义结构体变量时，一定要进行初始化
xiaoming := Student{"dongge", true, 18, "shandong", 100}

方式3

// 定义结构体变量时，一定要进行初始化
xiaoming := Student{name: "dongge", sex: true, weight: 100}

注意：

  • 顺序初始化，每个成员必须初始化，在初始化时，值的顺序与结构体成员的顺序保持一致。

5. 赋值、取值

package main

import "fmt"

func main()  {
    // 定义一个结构体类型
    type Student struct {
        id int
        name string
        sex bool
        age int
        address string
        weight int
    }

    // 定义结构体变量时，一定要进行初始化
    xiaoming := Student{name: "dongge", sex: true, weight: 100}

    // 给结构体成员变量赋值
    xiaoming.id = 101
    xiaoming.age = 19
    xiaoming.address = "shandong"

    // 获取结构体成员变量的值
    fmt.Println(xiaoming.id)
    fmt.Println(xiaoming.name)
    fmt.Println(xiaoming.sex)
    fmt.Println(xiaoming.age)
    fmt.Println(xiaoming.address)
    fmt.Println(xiaoming.weight)
}

6. 比较、赋值整体

7. 结构体当做参数

结构体作为函数参数进行传递，是值传递

package main

import "fmt"

// 定义一个结构体类型
type Student struct {
    id int
    name string
    sex bool
    age int
    address string
    weight int
}

func Test(temp Student) {

    temp.name = "王老师"

    // 打印结果
    fmt.Println("Test函数中,", temp)
}

func main()  {
    // 定义结构体变量时，一定要进行初始化
    xiaoming := Student{101, "dongge", true, 18, "shandong", 100}

    // 调用函数，将结构体传递过去
    Test(xiaoming)

    // 打印结果
    fmt.Println("main函数中,", xiaoming)
}

8. 结构体数组

package main

import "fmt"

// 定义一个结构体类型
type Student struct {
    id int
    name string
    sex bool
    age int
    address string
    weight int
}

func main()  {
    students := []Student{
        {101, "王老师", true, 18, "shandong", 100},
        {102, "李老师", true, 19, "shandong", 70},
        {103, "赵老师", false, 20, "shandong", 90},
        {104, "张老师", true, 21, "shandong", 80},
    }

    // 使用for循环遍历数组
    for i, temp := range students{
        fmt.Println("i =", i, "结构体 =", temp)
    }
}
