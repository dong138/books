        # 提取出验证码
        captcha = session.get("captcha")
        try:
            # 为了防止用户输入大小写，影响判断，统一转换为小写
            if captcha.lower() != verify.lower():
                ret = {
                    "status": 5,
                    "errmsg": "验证码不正确"
                }
                return jsonify(ret)
        except Exception as ret:
            ret = {
                "status": 6,
                "errmsg": "哦哦，服务器小问题产生"
            }
            return jsonify(ret)
