2.3 完善业务

2.3.1 修改视图函数register


@index_blu.route("/register", methods=["GET", "POST"])
def register():
    if request.method == "POST":
        # print(request.form)
        # 提取 数据
        email = request.form.get("email")
        user_name = request.form.get("username")
        password = request.form.get("password")
        verify = request.form.get("verify")  # 后面再使用

        # 只要有1个需要的数据，没有，那么就返回数据有误
        if not (email and user_name and password):
            # 返回对应的数据
            ret = {
                "status": 2,
                "errmsg": "输入数据有误，请重新输入"
            }
            return jsonify(ret)

        # 业务处理：注册用户
        # 判断是否注册过
        # 1. 业务处理
        db_session = sessionmaker(bind=engine)()  # 生成链接数据库的实例
        user_ret = db_session.query(User).filter(or_(User.email == email, User.user_name == user_name)).first()
        if user_ret:
            # 2. 如果邮箱或者用户名已经存在，则不允许使用
            # 3. 返回对应的数据
            ret = {
                "status": 1,
                "errmsg": "邮箱或用户名已存在，请修改"
            }
        else:
            # 3. 未注册，那么则进行注册
            user_id = email[:email.find("@")]
            new_user = User(email=email, user_id=user_id, password=password, user_name=user_name)
            db_session.add(new_user)
            db_session.commit()

            # 3. 返回对应的数据
            ret = {
                "status": 0,
                "errmsg": "注册成功"
            }
        db_session.close()
        return jsonify(ret)
    elif request.method == "GET":
        return render_template("index/register.html")

2.3.2 完善register.html

    $(".login-btn").click(function(){
        // 获取数据
        var email = $("#email").val();
        var username = $("#user").val();
        var password = $("#password").val();
        var verify = $("#verify").val();

        $.ajax({
            url: "{{ url_for('.register') }}",
            type: 'post',
            {#jsonp: 'jsonpcallback',#}
            {#jsonpCallback: "flightHandler",#}
            {#async: false,#}
            data: {
                'email':email,
                'username':username,
                'password':password,
                'verify':verify
            },
            success: function(data){
                status = data.status;
                errmsg = data.errmsg;
                layer.msg(errmsg);
                if(status==0){
                    setTimeout(function(){
                        window.location.href = "{{ url_for('.index') }}";
                    }, "2000");
                }
            }
        })
    });
