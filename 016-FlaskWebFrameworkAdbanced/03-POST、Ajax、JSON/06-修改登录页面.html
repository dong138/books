3.4 给login.html添加ajax功能

下面的代码是从register.html中复制过来的，然后修改了url到login

    $(".login-btn").click(function(){
        // 获取数据
        var email = $("#email").val();
        var password = $("#password").val();
        var verify = $("#verify").val();

        $.ajax({
            url: "{{ url_for('.login') }}",
            type: 'post',
            {#jsonp: 'jsonpcallback',#}
            {#jsonpCallback: "flightHandler",#}
            {#async: false,#}
            data: {
                'email':email,
                'password':password,
                'verify':verify
            },
            success: function(data){
                status = data.status;
                errmsg = data.errmsg;
                layer.msg(errmsg);
                if(status==0){
                    setTimeout(function(){
                        window.location.href = "{{ url_for('.index') }}";
                    }, "2000");
                }
            }
        })
    });

3.5 重构视图函数login

@index_blu.route("/login", methods=["GET", "POST"])
def login():
    if request.method == "POST":
        email = request.form.get("email")
        password = request.form.get("password")

        if not (email and password):
            ret = {
                "status": 4,
                "errmsg": "缺少参数"
            }
            return jsonify(ret)

        print("提取出来的email是：", email)
        print("提取出来的密码是：", password)

        # 1. 业务处理
        db_session = sessionmaker(bind=engine)()  # 生成链接数据库的实例
        user_ret = db_session.query(User).filter(User.email == email, User.password == password).first()
        db_session.close()
        print(user_ret)
        if user_ret:
            # response = make_response(redirect(url_for(".index")))
            session["login_flag"] = "success"
            session["user_id"] = user_ret.user_id
            # return response
            ret = {
                "status": 0,
                "errmsg": "登录成功"
            }
        else:
            ret = {
                "status": 2,
                "errmsg": "用户名或密码不正确"
            }

        return jsonify(ret)

    else:
        return render_template("index/login.html")
